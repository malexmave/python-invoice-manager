CREATE TABLE articles(
  ID INTEGER PRIMARY KEY AUTOINCREMENT,
  article_no INTEGER NOT NULL,
  name TEXT NOT NULL,
  description TEXT,
  type TEXT,
  tax_rate INT NOT NULL,
  gross DECIMAL(16,2),
  comment TEXT,
  last_modified INTEGER NOT NULL,
  active BOOLEAN DEFAULT TRUE
);
